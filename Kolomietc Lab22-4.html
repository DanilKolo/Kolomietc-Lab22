function calculate(a, b, c) { // Обчислення середнього арифметичного let average = (a + b + c) / 3; // Виведення результату на екран console.log(`Середнє арифметичне чисел ${a}, ${b} і ${c} дорівнює ${average}`); } // Приклад використання: calculate(10, 15, 20); // Виведе: Середнє арифметичне чисел 10, 15 і 20 дорівнює 15 calculate(5, 0, -5); // Виведе: Середнє арифметичне чисел 5, 0 і -5 дорівнює 0
