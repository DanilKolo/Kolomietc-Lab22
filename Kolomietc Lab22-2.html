function perimeter(side, count) { if (side <= 0 || count <= 2) { return 'Неправильні вхідні дані. Сторона має бути більше 0 і кількість сторін має бути більше двох.'; } return side * count; } // Приклад використання: console.log(perimeter(5, 4)); // Поверне 20 (периметр квадрата зі стороною 5) console.log(perimeter(3, 6)); // Поверне 18 (периметр шестикутника зі стороною 3) console.log(perimeter(8, 3)); // Поверне 24 (периметр рівностороннього трикутника зі стороною 8)
